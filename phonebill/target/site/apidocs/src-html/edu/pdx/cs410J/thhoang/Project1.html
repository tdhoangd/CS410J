001package edu.pdx.cs410J.thhoang;
002
003
004import edu.pdx.cs410J.AbstractPhoneBill;
005import edu.pdx.cs410J.AbstractPhoneCall;
006
007import java.io.*;
008import java.lang.String;
009import java.util.ArrayList;
010
011/**
012 * The main class for the CS410J Phone Bill Project 1
013 */
014public class Project1 {
015
016    public static void main(String[] args) {
017        String customer = null;
018        String callerNumber = null;
019        String calleeNumber = null;
020        String startDate = null;
021        String startTime = null;
022        String endDate = null;
023        String endTime = null;
024        ArrayList<String> options = new ArrayList<String>();
025        ArrayList<String> arguments = new ArrayList<String>();
026
027        if (args.length == 0) {
028            printErrMessageAndExit("Missing command line arguments");
029        }
030
031        for (String arg : args) {
032            if (arg.charAt(0) == '-')
033                options.add(arg);
034            else
035                arguments.add(arg);
036        }
037
038        for (String opt : options) {
039            if (opt.equals("-README")) {
040                printReadMeAndExit();
041            }
042        }
043
044        for (String arg : arguments) {
045
046            if (customer == null) {
047                customer = arg;
048            } else if (callerNumber == null) {
049                callerNumber = arg;
050            } else if (calleeNumber == null) {
051                calleeNumber = arg;
052            } else if (startDate == null) {
053                startDate = arg;
054            } else if (startTime == null) {
055                startTime = arg;
056            } else if (endDate == null) {
057                endDate = arg;
058            } else if (endTime == null) {
059                endTime = arg;
060            }
061        }
062
063        // print err message and exit if one of argument is missing
064        if (customer == null) {
065            printErrMessageAndExit("Missing customer");
066        } else if (callerNumber == null) {
067            printErrMessageAndExit("Missing callerNumber");
068        } else if (calleeNumber == null) {
069            printErrMessageAndExit("Missing calleeNumber");
070        } else if (startDate == null) {
071            printErrMessageAndExit("Missing startDate");
072        } else if (startTime == null) {
073            printErrMessageAndExit("Missing startTime");
074        } else if (endDate == null) {
075            printErrMessageAndExit("Missing endDate");
076        } else if (endTime == null) {
077            printErrMessageAndExit("Missing endTime");
078        }
079
080        // validate arguments format
081        if (!checkPhoneNumber(callerNumber)) {
082            printErrMessageAndExit("CallerNumber is not input in the right format: nnn-nnn-nnnn (where n is 0-9)");
083        } else if (!checkPhoneNumber(calleeNumber)) {
084            printErrMessageAndExit("CalleeNumber is not input is the right format: nnn-nnn-nnnn (where n is 0-9)");
085        } else if (!checkDateFormat(startDate)) {
086            printErrMessageAndExit("startDate is not input in the right format: mm/dd/yyyy");
087        } else if (!checkTimeFormat(startTime)) {
088            printErrMessageAndExit("startTime is not input in the right format: hh:mm");
089        } else if (!checkDateFormat(endDate)) {
090            printErrMessageAndExit("endDate is not input in the right format: mm/dd/yyyy");
091        } else if (!checkTimeFormat(endTime)) {
092            printErrMessageAndExit("endTime is not input in the right format: hh:mm");
093        }
094
095        // add new phone call to collection
096        AbstractPhoneBill newBill;
097        AbstractPhoneCall aCall;
098
099        try {
100            String start = startDate + " " + startTime;
101            String end = endDate + " " + endTime;
102
103            newBill = new PhoneBill(customer);
104            aCall = new PhoneCall(callerNumber, calleeNumber, start, end);
105            newBill.addPhoneCall(aCall);
106
107            System.out.println("Successful added new phone call");
108
109            for (String s : options) {
110                if (s.equals("-print")) {
111                    System.out.println("Description of new phone call: ");
112                    System.out.println(aCall.toString());
113                }
114            }
115
116        } catch (Exception e) {
117            printErrMessageAndExit("Fail to add new phone call");
118        }
119
120
121        System.exit(1);
122    }
123
124    /**
125     * Validate a date is in mm/dd/yyyy or not
126     * @param date
127     * @return boolean
128     */
129    private static boolean checkDateFormat(String date) {
130
131        if (date.matches("\\d{2}/\\d{2}/\\d{4}"))
132            return true;
133        else if (date.matches("\\d{1}/\\d{2}/\\d{4}"))
134            return true;
135        else if (date.matches("\\d{2}/\\d{1}/\\d{4}"))
136            return true;
137        else if (date.matches("\\d{1}/\\d{1}/\\d{4}"))
138            return true;
139        else
140            return false;
141
142    }
143
144    /**
145     * Validate if time is in hh:mm format or not
146     * @param time
147     * @return boolean
148     */
149    private static boolean checkTimeFormat(String time) {
150
151        if (time.matches("\\d{2}:\\d{2}"))
152            return true;
153        else if (time.matches("\\d{2}:\\d{1}")) {
154            return true;
155        } else if (time.matches("\\d{1}:\\d{2}")) {
156            return true;
157        } else if (time.matches("\\d{1}:\\d{1}")) {
158            return true;
159        } else {
160            return false;
161        }
162    }
163
164    /**
165     * Validate if the phone number is in the format nnn-nnn-nnnn
166     * @param phoneNumberString
167     * @return boolean
168     */
169    private static boolean checkPhoneNumber(String phoneNumberString) {
170
171        if (phoneNumberString.matches("\\d{3}-\\d{3}-\\d{4}"))
172            return  true;
173        else
174            return false;
175
176    }
177
178    /**
179     * Print out README.txt file and exit program.
180     */
181    private static void printReadMeAndExit() {
182        File f = new File("README.txt");
183
184        try {
185            if (!f.exists())
186                f.createNewFile();
187            FileReader fr = new FileReader(f);
188            BufferedReader br = new BufferedReader(fr);
189
190            StringBuffer sb = new StringBuffer();
191            String line = br.readLine();
192
193            while (line != null) {
194                sb.append(line);
195                sb.append("\n");
196                line = br.readLine();
197            }
198
199            System.out.print(sb.toString());
200            System.exit(1);
201
202        } catch (IOException e) {
203            e.printStackTrace();
204            System.err.println(" ** " + e);
205            System.exit(1);
206        }
207    }
208
209    /**
210     * Print out error message and exit program.
211     * @param message
212     */
213    private static void printErrMessageAndExit(String message) {
214        System.err.println(message);
215        System.exit(1);
216    }
