045    public void testPhoneBill() {
046        PhoneCall call1 = new PhoneCall("000-111-2222", "000-222-3333", "1/15/2015 10:10", "1/15/2015 10:30");
047        PhoneCall call2 = new PhoneCall("000-333-2222", "000-333-5555", "1/20/2015 14:22", "120/2015 15:99");
048
049        PhoneBill bill = new PhoneBill("A cat");
050        bill.addPhoneCall(call1);
051        bill.addPhoneCall(call2);
052
053        PhoneBill bill2 = new PhoneBill("A dog");
054        bill2.addPhoneCall(call1);
055        bill2.addPhoneCall(call2);
056
057        assertThat("Test customer name", bill.getCustomer(), is("A cat"));
058        assertThat("Test phone bill size", bill.getPhoneCalls().size(), is(2));
059        assertThat(bill.getPhoneCalls(), is(bill2.getPhoneCalls()));
060
061        System.out.println(bill.toString());
062        System.out.println(bill2.toString());
063
064
065    }

067    @Test
068    public void testPhoneCall() {
069        String callerNo = "000-111-2222";
070        String calleeNo = "000-222-3333";
071        String starTime = "1/15/2015 19:00";
072        String endTime = "1/15/2015 19:30";
073
074        PhoneCall phonecall = new PhoneCall(callerNo, calleeNo,starTime, endTime );

076        assertThat(phonecall.getCaller(), is("000-111-2222"));
